2. JSON DATA SCHEMA
------------------------------------------------------------------------
The microcontroller must write data as a single JSON object to the 
'/readings/latest' key. The parameters are case-sensitive and must match 
this schema:

{
  "temperature": 34.5,
  "pressure": 977,
  "altitude": 305,
  "food_level": 35,
  "cattle_present": 1,
  "health_status": "Normal",
  "visit_count": 7,
  "feeding_duration": 720,
  "object_distance": 45.2,
  "timestamp": "2026-05-30 02:15:00"
}

Fields Description:
* temperature (Float): Ambient temperature value in Celsius (°C).
* pressure (Integer): Atmospheric pressure in hPa.
* altitude (Integer): Elevation above sea level in meters.
* food_level (Integer): Remaining food percentage (e.g. 35 means 35% remaining).
* cattle_present (Integer): 1 if cattle are present at the station, 0 if absent.
* health_status (String): Health diagnostic status ("Normal", "Warning", "Critical").
* visit_count (Integer): Cumulative count of cattle visits today.
* feeding_duration (Integer): Total cumulative feeding duration in seconds.
* object_distance (Float): Measured distance to object/cattle in centimeters (cm).
* timestamp (String): Formatted date and time string of the sensor reading.

3. ESP8266 IMPLEMENTATION TEMPLATE (ARDUINO C++)
------------------------------------------------------------------------
To send this data using an ESP8266 and the 'FirebaseESP8266' library 
by Mobizt:

#include <ESP8266WiFi.h>
#include <FirebaseESP8266.h>

#define WIFI_SSID "YOUR_WIFI_SSID"
#define WIFI_PASSWORD "YOUR_WIFI_PASSWORD"

#define FIREBASE_HOST "iot-project-29231-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "AIzaSyANqn5SNvO9ZGuUzuT15aDqh3tc6qH8Mn4"

FirebaseData firebaseData;
FirebaseJson json;

void setup() {
  Serial.begin(115200);

  // WiFi Connection
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected.");

  // Firebase Init
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  Firebase.reconnectWiFi(true);
}

void loop() {
  // Read sensor values (Replace with actual physical sensor readings)
  float tempVal = 34.5;               // Read from temperature sensor
  float pressVal = 977;               // Read from barometric sensor in hPa
  float altVal = 305;                 // Read from altitude calculation
  int foodLevelVal = 35;              // Read from ultrasonic sensor (remaining food percentage)
  int cattlePresentVal = 1;           // PIR/ultrasonic detection (1 = Present, 0 = Absent)
  String healthStatusVal = "Normal";  // Computed health status from sensors
  int visitCountVal = 7;              // Total visits tracked by the microcontroller
  int feedingDurationVal = 720;       // Cumulative feeding time in seconds
  float objDistanceVal = 45.2;        // Ultrasonic distance measurement in cm
  String timestampVal = "2026-05-30 02:15:00"; // Formatted timestamp

  // Load readings into JSON payload structure
  json.clear();
  json.add("temperature", tempVal);
  json.add("pressure", pressVal);
  json.add("altitude", altVal);
  json.add("food_level", foodLevelVal);
  json.add("cattle_present", cattlePresentVal);
  json.add("health_status", healthStatusVal);
  json.add("visit_count", visitCountVal);
  json.add("feeding_duration", feedingDurationVal);
  json.add("object_distance", objDistanceVal);
  json.add("timestamp", timestampVal);

  // Push to Firebase Realtime Database
  if (Firebase.setJSON(firebaseData, "/readings/latest", json)) {
    Serial.println("Data pushed successfully.");
  } else {
    Serial.print("Failed to push data: ");
    Serial.println(firebaseData.errorReason());
  }

  delay(5000); // Send data updates every 5 seconds
}
