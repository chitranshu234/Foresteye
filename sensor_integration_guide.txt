------------------------------------------------------------------------
2. JSON DATA SCHEMA
------------------------------------------------------------------------
The microcontroller must write data as a single JSON object to the 
'/readings/latest' key. The parameters are case-sensitive and must match 
this schema:

{
  "temperature": 34.64,
  "pressure": 97530.11,
  "ir": 1,
  "food_cm": 0,
  "altitude": 320.85
}

Fields Description:
* temperature (Float): Ambient temperature value in Celsius (°C).
* pressure (Float): Atmospheric pressure in Pascals (Pa). *Website automatically converts this to hPa.*
* ir (Integer): Infrared status (maps to rain status). Send 0 for wet/rain, and 1 for clear/dry.
* food_cm (Integer): Height in cm representing food level / empty space.
* altitude (Float): Elevation above sea level in meters.

------------------------------------------------------------------------
3. ESP8266 IMPLEMENTATION TEMPLATE (ARDUINO C++)
------------------------------------------------------------------------
To send this data using an ESP8266 and the 'FirebaseESP8266' library 
by Mobizt:

#include <ESP8266WiFi.h>
#include <FirebaseESP8266.h>

#define WIFI_SSID "YOUR_WIFI_SSID"
#define WIFI_PASSWORD "YOUR_WIFI_PASSWORD"

#define FIREBASE_HOST "iot-project-29231-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "AIzaSyANqn5SNvO9ZGuUzuT15aDqh3tc6qH8Mn4"

FirebaseData firebaseData;
FirebaseJson json;

void setup() {
  Serial.begin(115200);

  // WiFi Connection
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected.");

  // Firebase Init
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  Firebase.reconnectWiFi(true);
}

void loop() {
  // Read sensor values (Replace with actual physical sensor readings)
  float tempVal = 34.64;     // Read from temperature sensor
  float pressVal = 97530.11; // Read from barometric sensor in Pascals
  int irVal = 1;             // Read from infrared obstacle/rain sensor (0 = wet, 1 = dry)
  int foodCmVal = 0;         // Read from ultrasonic sensor (distance to food in cm)
  float altVal = 320.85;     // Read from altitude calculation

  // Load readings into JSON payload structure
  json.clear();
  json.add("temperature", tempVal);
  json.add("pressure", pressVal);
  json.add("ir", irVal);
  json.add("food_cm", foodCmVal);
  json.add("altitude", altVal);

  // Push to Firebase Realtime Database
  if (Firebase.setJSON(firebaseData, "/readings/latest", json)) {
    Serial.println("Data pushed successfully.");
  } else {
    Serial.print("Failed to push data: ");
    Serial.println(firebaseData.errorReason());
  }

  delay(5000); // Send data updates every 5 seconds
}
